/**
 * @file commands/interaction/kiss.js
 * @description Cette commande permet de faire un bisous à une cible
 * @author gelo
 * @module commands/kiss
 */

const { SlashCommandBuilder } = require("@discordjs/builders");
const { social } = require("../../utils/naofumi");

module.exports = {

    data: new SlashCommandBuilder()
        .setName("kiss")
        .setDescription("Faire un bisous à un utilisateur")
        .addUserOption(option =>
            option
                .setName("target")
                .setDescription("La personne que vous souhaitez embrasser")
        ),

    execute: async (interaction, client) =>
    {

        /**
         * @function createKissEmbed
         * @description Retourne un embed contenant un gif de bisous
         * @returns {Embed}
         */
        async function createKissEmbed()
        {
            const target = interaction.options.getUser("target") || client.user;
            return await social.create({
                interactionName: interaction.commandName,
                author: interaction.user,
                title: `${ interaction.user.username } fait un bisous à ${ target.username }`
            });
        }


        interaction.reply({ embeds: [await createInteractionEmbed()] });

    }

}
