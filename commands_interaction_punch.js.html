/**
 * Cette commande permet de frapper une cible
 * 
 * @author gelo
 * @module commands/punch
 */

const { SlashCommandBuilder } = require("@discordjs/builders");
const { social } = require("../../utils/naofumi");

module.exports = {

    data: new SlashCommandBuilder()
        .setName("hug")
        .setDescription("Faire un câlin à un utilisateur")
        .addUserOption(option =>
            option
                .setName("target")
                .setDescription("La personne que vous souhaitez câliner")
        ),

    execute: async (interaction, client) =>
    {

        /**
         * Retourne un embed contenant un punch gif
         *
         *  @function createPunchEmbed
         * @returns {Embed}
         */
        async function createPunchEmbed()
        {
            const target = interaction.options.getUser("target") || client.user;
            return await social.create({
                interactionName: interaction.commandName,
                author: interaction.user,
                title: `${ interaction.user.username } frappe ${ target.username }`
            });
        }

        interaction.reply({ embeds: [await createPunchEmbed()] });

    }

}
