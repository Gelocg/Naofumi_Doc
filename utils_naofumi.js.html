/**
 * @function createSocial
 * @description Retourne un embed différent en fonction du nom de l'interaction
 * @param {{interactionName, author, title}} param0 
 * @returns {Embed}
 */

async function createSocial({ interactionName = "", author, title = "" })
{
    const interaction = await api.get.gif(interactionName)
    return halyma.embed({
        author: [title, author.displayAvatarURL()],
        image: interaction.url,
        description: `Vous ne voyez pas l'image ? [Cliquez ici](${ interaction.url })`
    });
}


/**
 * @class module:utils.Error
 * @classdesc Multiples méthodes pour récupérer des embeds d'erreurs
 */

class Error
{

    /**
     * @method module:utils.Error#noperm
     * @returns {Embed}
     */

    static noperm()
    {
        return halyma.embed({
            color: "RED",
            title: ":warning: Erreur",
            description: "Vous n'avez pas la permission de faire ceci"
        })
    }


    /**
     * @method module:utils.Error#custom
     * @param {string} error 
     * @returns {Embed}
     */

    static custom(error = "")
    {
        return halyma.embed({
            color: "RED",
            title: ":warning: Erreur",
            description: error
        })
    }


}
