/**
 * @file events/interactionCreate.js
 * @description Détecter quand une interaction est créé
 * @author gelo
 * @module commands/kiss
 */

const { error } = require("../utils/naofumi");

module.exports = async (client, interaction) =>
{

    /**
     * @function executeCommand
     * @description Execute la commande si l'interaction en est une
     */
    function executeCommand()
    {
        if (!interaction.isCommand()) return;

        const command = client.commands.get(interaction.commandName);
        if (!command) return;

        if (command.permission && !interaction.member.hasPermission(command.permission))
            return interaction.reply({
                ephemeral: true,
                embeds: [error.noperm()]
            });

        command.execute(interaction, client);
    }

    executeCommand();

}
