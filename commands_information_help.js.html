/**
 * @file commands/information/help.js
 * @description Cette commande permet de récupérer la liste des commandes et leurs description
 * @author gelo
 * @module commands/help
 */

const { SlashCommandBuilder } = require("@discordjs/builders");
const { halyma } = require("halymafunctions");

module.exports = {

    data: new SlashCommandBuilder()
        .setName("help")
        .setDescription("Récupérer la liste des commandes"),

    execute: async (interaction, client) =>
    {

        /**
         * @function generateHelpDescription
         * @description Retourne la description pour l'embed d'aide.
         * La description contient toutes les commandes au format
         * command.name command.options : command.description
         * @returns {string}
         */
        function generateHelpDescription()
        {

            let helpDescription = "";
            client.commands.map(a => a)
                .forEach((command) =>
                {
                    helpDescription += `➥ ${ command.data.name }  ${ command.data.options[0] ? "\`" + command.data.options.map(a => a.name).join("` `") + "\`" : "" } : ${ command.data.description }\n`
                })

            return helpDescription;

        }

        /**
         * @function generateHelpEmbed
         * @description Retourne un Discord.embed contenant la liste des commandes et leurs description
         * @returns {Embed}
         */
        function generateHelpEmbed()
        {

            return halyma.embed({
                author: [
                    `Page d'aide demandée par ${ interaction.user.tag }`,
                    interaction.user.displayAvatarURL()
                ],
                description: generateHelpDescription()
            })

        }


        interaction.reply({
            embeds: [generateEmbed()]
        })

    }

}
